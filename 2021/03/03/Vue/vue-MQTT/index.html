vue中连接MQTT服务器

vue 通讯
前端 vue
发布日期:   2021-03-03
文章字数:   638

        MQTT是一个基于客户端-服务器的消息发布/订阅传输协议。MQTT协议是轻量、简单、开放和易于实现的，这些特点使它适用范围非常广泛。在很多情况下，包括受限的环境中，如：机器与机器（M2M）通信和物联网（IoT）。其在，通过卫星链路通信传感器、偶尔拨号的医疗设备、智能家居、及一些小型化设备中已广泛使用。

在vue项目中连接MQTT服务器

1.命令行安装mqtt

npm install mqtt --save

2.项目中导入mqtt

import mqtt from 'mqtt'

3.使用
<template>
  <div>
    <van-nav-bar title="MQTT服务器连接测试" left-arrow />
    <van-button type="primary" size="small" @click="connectServer" block>
      点击连接mqtt服务器
    </van-button>
    <van-button type="danger" size="small" @click="endServer" block>
      点击断开
    </van-button>
    <van-button type="info" size="small" @click="sendContent" block>
      发送信息
    </van-button>
    <van-popup round closeable position="bottom" v-model="showSubscribe">
      <van-cell-group style="margin: 10px 0">
        <van-field v-model="subscribe1" label="订阅号" placeholder="请输入订阅号" />
        <van-field v-model="username" label="用户名" placeholder="请输入用户名" />
        <van-field v-model="url" label="连接地址" placeholder="请输入地址Url" />
      </van-cell-group>
      <van-button type="info" size="small" @click="connect" block>点击连接</van-button>
    </van-popup>
    <van-popup round closeable position="bottom" v-model="showContent">
      <van-cell-group>
        <van-field v-model="subscribe2" label="订阅号" placeholder="请输入订阅号" />
        <van-field
          v-model="message"
          label="信息"
          rows="3"
          autosize
          type="textarea"
          maxlength="50"
          placeholder="请输入要发送的信息"
          show-word-limit
        />
      </van-cell-group>
      <van-button type="info" size="small" @click="send" block>点击发送</van-button>
    </van-popup>
  </div>
</template>
<script>
import mqtt from 'mqtt';
export default {
  data() {
    return {
      client: null,
      subscribe1: '',
      subscribe2: '',
      username: '',
      url: 'ws://address:port', // ip地址和端口
      message: '',
      showSubscribe: false,
      showContent: false,
    }
  },
  methods: {
    connectServer() {
      this.showSubscribe = true
    },
    connect(){
      console.log("连接中。。")
      var _this = this;
      // 连接mqtt服务器
      this.client = mqtt.connect(this.url, {
        connectTimeout: 5000,
        clientId: '小赵老弟',
        username: this.username,
        // password: 'zzdy123.',
        clean: true
      })
      // 连接成功之后的监听
      this.client.on('connect', (e) => {
        console.log("连接成功！！！", e)
        // 订阅主题
        _this.client.subscribe(this.subscribe1, {qos: 0}, (error) => { 
          if (!error) {
            console.log('订阅成功')
            this.showSubscribe = false
          } else {
            console.log('订阅失败')
            _this.client.end()
          }
        })
      })
      // 接收消息处理
      this.client.on('message', (topic, res) => {
        // console.log('收到来自', topic, '的消息', JSON.stringify(res))
        const resData = res.toString()
        // 处理接收到的数据
        console.log('接收到消息',resData)
      })
      this.client.on('close',(e) => {
        console.log("断开连接！！！", e)
      })
    },
    sendContent() {
      this.message = ''
      this.showContent = true;
    },
    send() {
      this.client.publish(this.subscribe2,this.message)
      this.showContent = false;
    },
    /** 断开连接 */
    endServer() {
      this.client.end()
    },
  },
}
</script>
